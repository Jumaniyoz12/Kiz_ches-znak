24   def read_csv_rows(path: str | Path) -> list[dict[str, str]]:
25       path = Path(path)
26       with path.open("r", encoding="utf-8-sig", newline="") as handle:
27           reader = csv.DictReader(handle)
28           return [{key: (value or "").strip() for key, value in row.items()} for row in reader if any(row.values())]

32       rows = read_csv_rows(path)

41   def merge_products(nomenclature_rows: list[dict[str, str]], gtin_rows: list[dict[str, str]]) -> list[Product]:
42       gtin_by_barcode: dict[str, str] = {}
43       color_by_barcode: dict[str, str] = {}
44       for row in gtin_rows:
45           barcode = only_digits(_get(row, "barcode"))
46           gtin = normalize_gtin(_get(row, "gtin"))
47           gtin_article = _get(row, "seller_article")
48           if barcode and gtin:
49               gtin_by_barcode[barcode] = gtin
50           if barcode:
51               color = _extract_color(gtin_article)
52               if color:
53                   color_by_barcode[barcode] = color
54   
55       products: list[Product] = []
56       for row in nomenclature_rows:
57           barcode = only_digits(_get(row, "barcode"))
58           if not barcode:
59               continue
60           product = _row_to_product(row)
61           products.append(
62               Product(
63                   brand=product.brand,
64                   subject=product.subject,
65                   seller_article=product.seller_article,
66                   wb_article=product.wb_article,
67                   size=product.size,
68                   color=product.color or color_by_barcode.get(barcode, ""),
69                   supplier=product.supplier or "ОсОО ТКЛ",
70                   barcode=barcode,
71                   gtin=gtin_by_barcode.get(barcode, ""),
72                   composition=product.composition,
73               )
74           )
75       return products

78   def build_label_items(products: list[Product], codes: list[str], validator: LocalCodeValidator) -> list[LabelItem]:
79       product_by_gtin = {normalize_gtin(product.gtin): product for product in products if product.gtin}
80       items: list[LabelItem] = []
81       unused_products = iter(products)
82   
83       for index, code in enumerate(codes, start=1):
84           code_gtin = normalize_gtin(extract_gtin(code))
85           product = product_by_gtin.get(code_gtin) if code_gtin else None
86           if product is None:
87               product = next(unused_products, None)
88           if product is None:
89               raise ValueError(f"Для кода #{index} не найден товар")
90           mark_code = validator.check_code(code, expected_gtin=product.gtin)
91           items.append(LabelItem(product=product, mark_code=mark_code, index=index))
92   
93       return items

96   def _row_to_product(row: dict[str, str]) -> Product:
97       return Product(
98           brand=_get(row, "brand"),
99           subject=_get(row, "subject"),
100          seller_article=_get(row, "seller_article"),
101          wb_article=only_digits(_get(row, "wb_article")),
102          size=_get(row, "size"),
103          color=_extract_color(_get(row, "seller_article")) or _get(row, "color"),
104          supplier=_get(row, "supplier"),
105          barcode=only_digits(_get(row, "barcode")),
106          gtin=normalize_gtin(_get(row, "gtin")),
107          composition=_get(row, "composition"),
108      )

111  def _get(row: dict[str, str], field: str) -> str:
112      for alias in FIELD_ALIASES[field]:
113          if alias in row and row[alias] is not None:
114              return str(row[alias]).strip()
115      return ""

118  def _extract_color(article: str) -> str:
119      marker = "ц:"
120      if marker not in article:
121          return ""
122      color = article.split(marker, 1)[1]
123      for separator in (";", ", р:", ",р:", " р:"):
124          color = color.split(separator, 1)[0]
125      return color.strip()

128  def only_digits(value: str) -> str:
129      return "".join(ch for ch in str(value or "") if ch.isdigit())
