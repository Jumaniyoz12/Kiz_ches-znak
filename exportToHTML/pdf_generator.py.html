90   def create_labels_pdf(items: list[LabelItem], output_path: str | Path, include_mark_code: bool = True) -> Path:
91       _register_fonts()
92       output_path = Path(output_path)
93       output_path.parent.mkdir(parents=True, exist_ok=True)
94   
95       canvas = Canvas(str(output_path), pagesize=landscape((LABEL_H, LABEL_W)))
96       for item in items:
97           _draw_label(canvas, item, include_mark_code=include_mark_code)
98           canvas.showPage()
99       canvas.save()
100      return output_path

103  def _draw_label(canvas: Canvas, item: LabelItem, include_mark_code: bool = True) -> None:
104      product = item.product
105      mark = item.mark_code
106  
107      canvas.setStrokeColor(colors.black)
108      canvas.setLineWidth(0.25)
109      canvas.rect(0.8 * mm, 0.8 * mm, LABEL_W - 1.6 * mm, LABEL_H - 1.6 * mm)
110  
111      _draw_centered_text(canvas, product.brand or "Timur Kids", TITLE_X + TITLE_W / 2, from_top(TITLE_Y, TITLE_H) - 0.2 * mm, 8.2, bold=True)
112  
113      if product.barcode:
114          barcode = createBarcodeDrawing(
115              "Code128",
116              value=product.barcode,
117              barHeight=BARCODE_H,
118              humanReadable=False,
119              width=BARCODE_W,
120              height=BARCODE_H,
121          )
122          renderPDF.draw(barcode, canvas, BARCODE_X, from_top(BARCODE_Y, BARCODE_H))
123          _draw_centered_text(canvas, product.barcode, BARCODE_NUMBER_X + BARCODE_NUMBER_W / 2, from_top(BARCODE_NUMBER_Y, BARCODE_NUMBER_H) + 0.1 * mm, 4.4)
124  
125      canvas.setFont(FONT_BOLD, 5.2)
126      subject_lines = _wrap_text(product.subject, PRODUCT_W, FONT_BOLD, 5.2, max_lines=2)
127      y = from_top(PRODUCT_Y, 1.7 * mm)
128      for line in subject_lines:
129          canvas.drawString(PRODUCT_X, y, line)
130          y -= 2.2 * mm
131  
132      _draw_label_value(canvas, "Артикул:", product.seller_article, ARTICLE_LABEL_X, ARTICLE_VALUE_X, ARTICLE_Y, ARTICLE_VALUE_W)
133      _draw_label_value(canvas, "Артикул WB:", product.wb_article, WB_LABEL_X, WB_VALUE_X, WB_Y, WB_VALUE_W)
134      _draw_label_value(canvas, "Размер:", product.size, SIZE_LABEL_X, SIZE_VALUE_X, SIZE_Y, SIZE_VALUE_W)
135      _draw_label_value(canvas, "Цвет:", product.color, COLOR_LABEL_X, COLOR_VALUE_X, COLOR_Y, COLOR_VALUE_W)
136      _draw_label_value(canvas, "Поставщик:", product.supplier, SUPPLIER_LABEL_X, SUPPLIER_VALUE_X, SUPPLIER_Y, SUPPLIER_VALUE_W, label_size=3.8, value_size=4.5)
137  
138      canvas.setFont(FONT_BOLD, 8.2)
139      canvas.drawCentredString(TOP_NUMBER_X + TOP_NUMBER_W / 2, from_top(TOP_NUMBER_Y, TOP_NUMBER_H) + 0.4 * mm, str(item.index))
140  
141      if include_mark_code and mark.raw:
142          _draw_mark_code(canvas, mark.raw)
143      else:
144          _draw_eac(canvas, EAC_X, from_top(EAC_Y, EAC_H), 5.8)

147  def _draw_mark_code(canvas: Canvas, code: str) -> None:
148      dm_y = from_top(DM_Y, DM_H)
149      quiet = 1.0 * mm
150      canvas.saveState()
151      canvas.setFillColor(colors.white)
152      canvas.setStrokeColor(colors.white)
153      canvas.rect(DM_X - quiet, dm_y - quiet, DM_W + quiet * 2, DM_H + quiet * 2, fill=1, stroke=0)
154      canvas.restoreState()
155  
156      dm_image = create_datamatrix_image(code)
157      canvas.drawImage(ImageReader(dm_image), DM_X, dm_y, width=DM_W, height=DM_H, mask="auto")
158  
159      canvas.saveState()
160      canvas.translate(VERTICAL_X, from_top(VERTICAL_Y, VERTICAL_H))
161      canvas.rotate(90)
162      canvas.setFont(FONT_REGULAR, 2.8)
163      canvas.drawString(0, 0, _fit_text(code, VERTICAL_H, FONT_REGULAR, 2.8))
164      canvas.restoreState()
165  
166      _draw_eac(canvas, EAC_X, from_top(EAC_Y, EAC_H), 5.8)

168  def create_datamatrix_image(code: str) -> Image.Image:
169      payload = prepare_marking_code_for_datamatrix(code)
170      encoded = encode_datamatrix(payload, scheme="Ascii", size="SquareAuto")
171      image = Image.frombytes("RGB", (encoded.width, encoded.height), encoded.pixels)
172      return image

175  def prepare_marking_code_for_datamatrix(code: str) -> bytes:
176      # Keep ASCII 29 group separators if they exist in the source KIZ.
177      # libdmtx expects bytes; utf-8 keeps digits and common Honest Sign symbols stable.
178      normalized = str(code or "").replace("\\x1d", "\x1d").replace("<GS>", "\x1d")
179      return normalized.encode("utf-8")

186  def _draw_label_value(
187      canvas: Canvas,

189      value: str,

208  def _draw_centered_text(canvas: Canvas, text: str, x: float, y: float, size: float, bold: bool = False) -> None:
209      font = FONT_BOLD if bold else FONT_REGULAR
210      text = _fit_text(text, LABEL_W - 4 * mm, font, size)
211      canvas.setFont(font, size)
212      canvas.drawCentredString(x, y, text)

215  def _wrap_text(text: str, max_width: float, font: str, size: float, max_lines: int) -> list[str]:
216      words = str(text or "").split()
217      if not words:
218          return [""]
219  
220      lines: list[str] = []
221      current = ""
222      used_words = 0
223      for word in words:
224          candidate = f"{current} {word}".strip()
225          if current and stringWidth(candidate, font, size) > max_width:
226              lines.append(current)
227              current = word
228              if len(lines) == max_lines - 1:
229                  used_words += 1
230                  break
231          else:
232              current = candidate
233              used_words += 1
234      if current and len(lines) < max_lines:
235          lines.append(current)
236      if used_words < len(words) and lines:
237          lines[-1] = _fit_text(lines[-1] + " " + " ".join(words[used_words:]), max_width, font, size)
238      return lines[:max_lines]

241  def _fit_text(text: str, max_width: float, font: str, size: float) -> str:
242      text = str(text or "")
243      if stringWidth(text, font, size) <= max_width:
244          return text
245      ellipsis = "..."
246      result = text
247      while result and stringWidth(result + ellipsis, font, size) > max_width:
248          result = result[:-1]
249      return result + ellipsis if result else ""

252  def _register_fonts() -> None:
253      if FONT_REGULAR in pdfmetrics.getRegisteredFontNames():
254          return
255      regular = Path("C:/Windows/Fonts/arial.ttf")
256      bold = Path("C:/Windows/Fonts/arialbd.ttf")
257      if regular.exists() and bold.exists():
258          pdfmetrics.registerFont(TTFont(FONT_REGULAR, str(regular)))
259          pdfmetrics.registerFont(TTFont(FONT_BOLD, str(bold)))
260          return
261      pdfmetrics.registerFont(TTFont(FONT_REGULAR, "DejaVuSans.ttf"))
262      pdfmetrics.registerFont(TTFont(FONT_BOLD, "DejaVuSans-Bold.ttf"))
