9    def read_google_sheet_csv(spreadsheet_url: str, sheet_name_or_gid: str) -> list[dict[str, str]]:
10       csv_url = build_csv_export_url(spreadsheet_url, sheet_name_or_gid)
11       request = urllib.request.Request(csv_url, headers={"User-Agent": "labelbot/1.0"})
12       with urllib.request.urlopen(request, timeout=30) as response:
13           text = response.read().decode("utf-8-sig")
14       reader = csv.DictReader(io.StringIO(text))
15       return [{key: (value or "").strip() for key, value in row.items()} for row in reader if any(row.values())]

18   def build_csv_export_url(spreadsheet_url: str, sheet_name_or_gid: str) -> str:
19       sheet_id = extract_spreadsheet_id(spreadsheet_url)
20       value = str(sheet_name_or_gid or "").strip()
21       if value.isdigit():
22           return f"https://docs.google.com/spreadsheets/d/{sheet_id}/export?format=csv&gid={value}"
23       encoded_sheet = urllib.parse.quote(value)
24       return f"https://docs.google.com/spreadsheets/d/{sheet_id}/gviz/tq?tqx=out:csv&sheet={encoded_sheet}"

27   def extract_spreadsheet_id(spreadsheet_url: str) -> str:
28       parts = urllib.parse.urlparse(spreadsheet_url)
29       path_parts = [part for part in parts.path.split("/") if part]
30       if "d" in path_parts:
31           index = path_parts.index("d")
32           if index + 1 < len(path_parts):
33               return path_parts[index + 1]
34       if spreadsheet_url and "/" not in spreadsheet_url:
35           return spreadsheet_url
36       raise ValueError("Не удалось понять ссылку Google Sheets")
