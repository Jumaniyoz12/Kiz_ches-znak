1    from __future__ import annotations
2    
3    import argparse
4    
5    from data import build_label_items, merge_products, read_codes, read_csv_rows, read_products
6    from google_sheets import read_google_sheet_csv
7    from pdf_generator import create_labels_pdf
8    from validator import LocalCodeValidator
9    
10   
11   def main() -> None:
12       parser = argparse.ArgumentParser(description="Generate 58x40 mm Honest Sign labels PDF.")
13       parser.add_argument("--products", help="Old mode: CSV file with ready products")
14       parser.add_argument("--nomenclature", help="CSV export of sheet 'Отчёт с перечнем номенклатур'")
15       parser.add_argument("--gtin", help="CSV export of sheet 'GTIN'")
16       parser.add_argument("--google-sheet-url", help="Google Sheets URL or spreadsheet id")
17       parser.add_argument("--nomenclature-sheet", default="Отчёт с перечнем номенклатур")
18       parser.add_argument("--gtin-sheet", default="GTIN")
19       parser.add_argument("--codes", required=True, help="TXT file with Honest Sign codes, one per line")
20       parser.add_argument("--out", required=True, help="Output PDF path")
21       args = parser.parse_args()
22   
23       if args.google_sheet_url:
24           nomenclature_rows = read_google_sheet_csv(args.google_sheet_url, args.nomenclature_sheet)
25           gtin_rows = read_google_sheet_csv(args.google_sheet_url, args.gtin_sheet)
26           products = merge_products(nomenclature_rows, gtin_rows)
27       elif args.nomenclature and args.gtin:
28           products = merge_products(read_csv_rows(args.nomenclature), read_csv_rows(args.gtin))
29       elif args.products:
30           products = read_products(args.products)
31       else:
32           raise SystemExit("Укажите --google-sheet-url или пару --nomenclature/--gtin")
33   
34       codes = read_codes(args.codes)
35       items = build_label_items(products, codes, LocalCodeValidator())
36       output = create_labels_pdf(items, args.out)
37       print(f"PDF готов: {output}")
38   
39   
40   if __name__ == "__main__":
41       main()
